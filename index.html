document.getElementById('passInput').addEventListener('keypress', function (e) { if (e.key === 'Enter') checkLogin(); }); // --- HARİTA VE VERİTABANI MANTIĞI --- function initApp() { const dbName = "GTAMapDB_Final"; let db; let savedData = []; let customImgBase64 = null; const map = L.map('map', { crs: L.CRS.Simple, minZoom: -4, maxZoom: 4 }); let currentMapLayer = null; let markersLayer = L.layerGroup().addTo(map); const request = indexedDB.open(dbName, 1); request.onupgradeneeded = (e) => { db = e.target.result; if (!db.objectStoreNames.contains("settings")) db.createObjectStore("settings"); if (!db.objectStoreNames.contains("markers")) db.createObjectStore("markers", { keyPath: "id" }); }; request.onsuccess = (e) => { db = e.target.result; const settingsTx = db.transaction("settings", "readonly").objectStore("settings"); const mapReq = settingsTx.get("mainMap"); mapReq.onsuccess = () => { if(mapReq.result) loadMap(mapReq.result); }; const markersTx = db.transaction("markers", "readonly").objectStore("markers"); const markersReq = markersTx.getAll(); markersReq.onsuccess = () => { savedData = markersReq.result || []; render(); }; }; function loadMap(src) { const img = new Image(); img.onload = function() { if (currentMapLayer) map.removeLayer(currentMapLayer); const bounds = [[0, 0], [this.height, this.width]]; currentMapLayer = L.imageOverlay(src, bounds).addTo(map); map.fitBounds(bounds); render(); }; img.src = src; } document.getElementById('mainMapInput').onchange = (e) => { const reader = new FileReader(); reader.onload = (ev) => { db.transaction("settings", "readwrite").objectStore("settings").put(ev.target.result, "mainMap"); loadMap(ev.target.result); }; reader.readAsDataURL(e.target.files[0]); }; document.getElementById('markerImgInput').onchange = (e) => { const reader = new FileReader(); reader.onload = (ev) => { customImgBase64 = ev.target.result; document.getElementById('typeSelect').value = 'custom_img'; }; reader.readAsDataURL(e.target.files[0]); }; map.on('click', (e) => { if (!currentMapLayer) return; const obj = { id: Date.now(), lat: e.latlng.lat, lng: e.latlng.lng, size: parseInt(document.getElementById('sizeSlider').value), type: document.getElementById('typeSelect').value, img: document.getElementById('typeSelect').value === 'custom_img' ? customImgBase64 : null }; db.transaction("markers", "readwrite").objectStore("markers").add(obj); savedData.push(obj); render(); }); function render() { markersLayer.clearLayers(); const isEdit = document.getElementById('editMode').checked; savedData.forEach(data => { const iconStyle = `width:100%; height:100%; background-image: ${data.img ? 'url('+data.img+')' : 'none'};`; const icon = L.divIcon({ className: `pixel-marker marker-${data.type}`, iconSize: [data.size, data.size], iconAnchor: [data.size/2, data.size/2], html: `
` }); const marker = L.marker([data.lat, data.lng], { icon: icon, draggable: isEdit }).addTo(markersLayer); marker.on('dragend', (ev) => { const found = savedData.find(d => d.id === data.id); found.lat = ev.target.getLatLng().lat; found.lng = ev.target.getLatLng().lng; db.transaction("markers", "readwrite").objectStore("markers").put(found); }); marker.on('contextmenu', () => { db.transaction("markers", "readwrite").objectStore("markers").delete(data.id); savedData = savedData.filter(d => d.id !== data.id); render(); }); }); } window.clearAllData = function() { if(confirm("DİKKAT: Tüm harita ve işaretler kalıcı olarak silinecek!")) { db.transaction("settings", "readwrite").objectStore("settings").clear(); db.transaction("markers", "readwrite").objectStore("markers").clear(); location.reload(); } } document.getElementById('editMode').onchange = render; }
